<script type="text/javascript">
    function closeShareFrame() {
        xn_content_share.closeShareView();
    }

    function closeInfoFrame() {
        xn_content_share.closeShareView();
    }

    (function($) {
        var isInfoFrameLoaded = false;

        window.xn_content_share = {
            shareBtnClick: function() {
                $('body').css('background-color', 'white');
                hidePlayer();
                $('#share_view').show();
                $('#share_view #share_frame').contents().find('.share_wrapper').show();
                $('#share_view').find('.share_iframe').get(0).contentWindow.xn_view_share.setFocus();
            },

            infoBtnClick: function() {
                var infoUrl = 'https://lcms.skku.edu/index.php?module=xn_view_page2013&act=dispXn_view_page2013MediaContentInfo&content_id=5e53960c4f61a';
                if (!isInfoFrameLoaded) {
                    var frameSrc = '<iframe id="info_frame" class="share_iframe" frameborder="0" src="' +
                        infoUrl + '"></iframe>';
                    $('#info_view').html(frameSrc);
                    isInfoFrameLoaded = true;
                }
                $('body').css('background-color', 'white');
                hidePlayer();
                $('#info_view').show();
            },

            closeShareView: function() {
                showPlayer();
                $('#share_view').hide();
                $('#info_view').hide();
                $('body').css('background-color', 'black');
                $(window).trigger("resize");
            }
        };

        function showPlayer() {
            $('#player-area').css('width', '100%');
            $('#player-area').css('height', '100%');
        }

        function hidePlayer() {
            $('#player-area').css('width', '0px');
            $('#player-area').css('height', '0px');
        }

    }(jQuery));
</script>
